Ukázka Python kódu


    import requests
    import json
    import os
    import platform
    from colorama import init, Fore, Style
    
    # Inicializace colorama
    init(autoreset=True)
    
    WEBHOOK_URL = "url"
    
    def clear_screen():
        if platform.system() == "Windows":
            os.system("cls")
        else:
            os.system("clear")
    
    def get_ip_info(ip):
        url = f"https://ipinfo.io/widget/demo/{ip}"
        response = requests.get(url)
        if response.status_code == 200:
            data = response.json()
            return data
        else:
            return {"error": "Unable to fetch data"}
    
    def format_webhook_message(ip_info):
        if "error" in ip_info:
            return {
                "content": ip_info["error"]
            }
        else:
            data = ip_info.get("data", {})
            privacy = data.get("privacy", {})
            vpn = privacy.get("vpn", False)
            proxy = privacy.get("proxy", False)
            tor = privacy.get("tor", False)
            relay = privacy.get("relay", False)
            hosting = privacy.get("hosting", False)
            service = privacy.get("service", "N/A")
    
            return {
                "content": f"IP Privacy Info:\n"
                           f"  VPN: {vpn}\n"
                           f"  Proxy: {proxy}\n"
                           f"  Tor: {tor}\n"
                           f"  Relay: {relay}\n"
                           f"  Hosting: {hosting}\n"
                           f"  Service: {service}"
            }
    
    def send_webhook(data):
        headers = {
            'Content-Type': 'application/json',
        }
        response = requests.post(WEBHOOK_URL, headers=headers, data=json.dumps(data))
        if response.status_code == 204:
            print("Webhook sent successfully.")
        else:
            print(f"Failed to send webhook: {response.status_code}, {response.text}")
    
    def print_ip_info(ip_info):
        if "error" in ip_info:
            print(ip_info["error"])
        else:
            input_ip = ip_info.get("input", "N/A")
            data = ip_info.get("data", {})
            ip = data.get("ip", "N/A")
            hostname = data.get("hostname", "N/A")
            city = data.get("city", "N/A")
            region = data.get("region", "N/A")
            country = data.get("country", "N/A")
            loc = data.get("loc", "N/A")
            org = data.get("org", "N/A")
            postal = data.get("postal", "N/A")
            timezone = data.get("timezone", "N/A")
    
            asn = data.get("asn", {})
            asn_number = asn.get("asn", "N/A")
            asn_name = asn.get("name", "N/A")
            asn_domain = asn.get("domain", "N/A")
            asn_route = asn.get("route", "N/A")
            asn_type = asn.get("type", "N/A")
    
            company = data.get("company", {})
            company_name = company.get("name", "N/A")
            company_domain = company.get("domain", "N/A")
            company_type = company.get("type", "N/A")
    
            privacy = data.get("privacy", {})
            vpn = privacy.get("vpn", False)
            proxy = privacy.get("proxy", False)
            tor = privacy.get("tor", False)
            relay = privacy.get("relay", False)
            hosting = privacy.get("hosting", False)
            service = privacy.get("service", "N/A")
    
            abuse = data.get("abuse", {})
            abuse_address = abuse.get("address", "N/A")
            abuse_country = abuse.get("country", "N/A")
            abuse_email = abuse.get("email", "N/A")
            abuse_name = abuse.get("name", "N/A")
            abuse_network = abuse.get("network", "N/A")
            abuse_phone = abuse.get("phone", "N/A")
    
            print(f"{Fore.GREEN}Input IP: {input_ip}{Style.RESET_ALL}")
            print(f"{Fore.GREEN}IP Address: {ip}{Style.RESET_ALL}")
            print(f"Hostname: {hostname}{Style.RESET_ALL}")
            print(f"City: {city}{Style.RESET_ALL}")
            print(f"Region: {region}{Style.RESET_ALL}")
            print(f"Country: {country}{Style.RESET_ALL}")
            print(f"{Fore.GREEN}Location: {loc}{Style.RESET_ALL}")
            print(f"{Fore.GREEN}Organization: {org}{Style.RESET_ALL}")
            print(f"{Fore.GREEN}Postal Code: {postal}{Style.RESET_ALL}")
            print(f"{Fore.GREEN}Timezone: {timezone}{Style.RESET_ALL}")
            print("\nASN Information:")
            print(f"  ASN: {asn_number}{Style.RESET_ALL}")
            print(f"  {Fore.GREEN}Name: {asn_name}{Style.RESET_ALL}")
            print(f"  {Fore.GREEN}Domain: {asn_domain}{Style.RESET_ALL}")
            print(f"  {Fore.GREEN}Route: {asn_route}{Style.RESET_ALL}")
            print(f"  {Fore.GREEN}Type: {asn_type}{Style.RESET_ALL}")
            print("\nCompany Information:")
            print(f"  Name: {company_name}{Style.RESET_ALL}")
            print(f"  Domain: {company_domain}{Style.RESET_ALL}")
            print(f"  Type: {company_type}{Style.RESET_ALL}")
            print("\nPrivacy Information:")
            print(f"  {Fore.RED}VPN: {vpn}{Style.RESET_ALL}")
            print(f"  {Fore.RED}Proxy: {proxy}{Style.RESET_ALL}")
            print(f"  {Fore.RED}Tor: {tor}{Style.RESET_ALL}")
            print(f"  {Fore.CYAN}Relay: {relay}{Style.RESET_ALL}")
            print(f"  {Fore.RED}Hosting: {hosting}{Style.RESET_ALL}")
            print(f"  {Fore.RED}Service: {service}{Style.RESET_ALL}")
            print("\nAbuse Contact:")
            print(f"  Address: {abuse_address}{Style.RESET_ALL}")
            print(f"  Country: {abuse_country}{Style.RESET_ALL}")
            print(f"  Email: {abuse_email}{Style.RESET_ALL}")
            print(f"  Name: {abuse_name}{Style.RESET_ALL}")
            print(f"  Network: {abuse_network}{Style.RESET_ALL}")
            print(f"  Phone: {abuse_phone}{Style.RESET_ALL}")
    
    def main():
        clear_screen()
        print(Fore.MAGENTA + r"""
     /$$$$$$ /$$$$$$$        /$$$$$$ /$$   /$$ /$$$$$$$$ /$$$$$$ 
    |_  $$_/| $$__  $$      |_  $$_/| $$$ | $$| $$_____//$$__  $$
      | $$  | $$  \ $$        | $$  | $$$$| $$| $$     | $$  \ $$
      | $$  | $$$$$$$/        | $$  | $$ $$ $$| $$$$$  | $$  | $$
      | $$  | $$____/         | $$  | $$  $$$$| $$__/  | $$  | $$
      | $$  | $$              | $$  | $$\  $$$| $$     | $$  | $$
     /$$$$$$| $$             /$$$$$$| $$ \  $$| $$     |  $$$$$$/
    |______/|__/            |______/|__/  \__/|__/      \______/ 
    
              Made by: ._methylphenidate
    """ + Style.RESET_ALL)
        ip = input("Zadejte IP adresu: ")
        clear_screen()
        ip_info = get_ip_info(ip)
        formatted_message = format_webhook_message(ip_info)
        send_webhook(formatted_message)
        print_ip_info(ip_info)
    
    if __name__ == "__main__":
        main()
